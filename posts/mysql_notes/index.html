MySQL Functions 12th NCERT

June 6, 2021

Functions available in MySQL

This is a compiled list of types of functions avaialable in mysql chapter of NCERT IP Class 12. Use this as a reference/basic manual. Do whatever you wish with it.

INTEGER FUNCTIONS –

SELECT POW(x,y);           -- power of x raise to y.
SELECT ROUND(num, n);      -- Rounds off num to n decimal places.
SELECT MOD(num1, num2);    -- Remainder of num1 divided by num1 or num1/num2.

STRING FUNCTIONS

SELECT LOWER(str);         -- Spits string in lowercase.
SELECT UPPER(str);         -- Spits string in uppercase.
SELECT SUBSTR(str, sp, n); -- Spits string starting from sp - starting position 
                           -- to n. If n not specified, It shows until end.
SELECT LENGTH(str);        -- Gives length of string as an int.
SELECT LEFT(str, n);       -- Spits n characters from the left of string.
SELECT RIGHT(str, n);      -- Spits n characters from the right of string.
SELECT INSTR(str, substr); -- Gives position of first occuerence 
                           -- of substr in str.
SELECT LTRIM(str);         -- Spits a string with leading whitespaces 
                           -- or whitespaces from left removed.
SELECT RTRIM(str);         -- Spits a string with trailing whitespaces 
                           -- or whitespaces from right removed.
SELECT TRIM(str);          -- Spits a string with all whitespaces removed 
                           -- from left and right.

DATE AND TIME FUNCTIONS

SELECT NOW();              -- Gives current date and time.
SELECT DATE(expression);   -- Gives date from the given expression.
                           -- Ex- SELECT DATE(NOW());
SELECT MONTH(date);        -- Returns month name from given date in integer form.
SELECT MONTHNAME(date);    -- Same as above but gives the actual name of month.
SELECT YEAR(date);         -- Gives the year from the given date.
SELECT DAY(date);          -- Gives the day from the given date in integer form.
SELECT DAYNAME(date);      -- Same as above but gives the actual name of day.

Aggregate Functions

SELECT MAX(col);          -- Returns the largest value from col (column).
SELECT MIN(col);          -- Returns the smallest value from col (column).
SELECT AVG(col);          -- Returns the average value from col (column).
SELECT SUM(col);          -- Returns the sum of all values in col (column).
SELECT COUNT(col);        -- Returns the number of values present 
                          -- in col (column) ignoring NULL Values.
SELECT COUNT(*);          -- Returns the number of records(rows) in Relation.
                          -- Can be used with WHERE to check for certain values.
